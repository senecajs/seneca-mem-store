function listents (si, entmap, qent, q, cb) {
  var list = []
  var canon = qent.canon$({
    object: true
  })
  var base = canon.base
  var name = canon.name
  var entset = entmap[base] ? entmap[base][name] : null
  if (entset) {
    _.keys(entset).forEach(function (id) {
      var ent = entset[id]
      for (var p in q) {
        if (!~p.indexOf('$') && q[p] !== ent[p]) {
          return
        }
      }
      ent = qent.make$(ent)
      list.push(ent)
    })
  }
